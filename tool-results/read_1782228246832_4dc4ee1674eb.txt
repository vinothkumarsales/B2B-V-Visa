    51→// Normalize document names — strip parenthetical annotations (e.g. "Bank Statement (min ₹3L)" → "Bank Statement")
    52→function normalizeDocName(name: string): string {
    53→  const cleaned = name.replace(/\s*\(.*?\)\s*/g, '').trim();
    54→  // Also normalize common variations
    55→  const map: Record<string, string> = {
    56→    'Hotel Booking Confirmation': 'Hotel Booking',
    57→    'Medical Insurance': 'Health Insurance',
    58→  };
    59→  return map[cleaned] || cleaned;
    60→}

   103→// Documents that are always covered by the passport upload section (not shown as additional)
   104→const ALWAYS_COVERED = new Set(['Passport', 'Photo']);
   105→
   106→function getRequiredAdditionalDocs(visaDocuments: string[]): { key: string; title: string; helper: string }[] {
   107→  const seen = new Set<string>();
   108→  return visaDocuments
   109→    .filter((doc) => !ALWAYS_COVERED.has(doc))
   110→    .map((doc) => {
   111→      const normalized = normalizeDocName(doc);
   112→      const meta = docNameToMeta[normalized] || docNameToMeta[doc];
   113→      if (meta) {
   114→        // Deduplicate by key in case two doc names normalize to same key
   115→        if (seen.has(meta.key)) return null;
   116→        seen.add(meta.key);
   117→        return { key: meta.key, title: doc, helper: meta.helper };
   118→      }
   119→      // Fallback for unknown doc names
   120→      const key = normalized.toLowerCase().replace(/[^a-z0-9]/g, '_');
   121→      if (seen.has(key)) return null;
   122→      seen.add(key);
   123→      return { key, title: doc, helper: `Upload ${doc}` };
   124→    })
   125→    .filter(Boolean) as { key: string; title: string; helper: string }[];
   126→}

   150→function formatDateForInput(dateStr: string): string {
   151→  if (!dateStr) return '';
   152→  // Handle DD/MM/YYYY format from OCR
   153→  const parts = dateStr.split(/[\/\-\.]/);
   154→  if (parts.length === 3) {
   155→    const [d, m, y] = parts.map((p) => p.trim());
   156→    // If year is 2 digits, convert to 4 digits
   157→    const fullYear = y.length === 2 ? (parseInt(y) > 50 ? '19' + y : '20' + y) : y;
   158→    return `${fullYear}-${m.padStart(2, '0')}-${d.padStart(2, '0')}`;
   159→  }
   160→  return '';
   161→}

   183→  const handlePassportUpload = useCallback(
   184→    async (e: React.ChangeEvent<HTMLInputElement>) => {
   185→      const file = e.target.files?.[0];
   186→      if (!file) return;
   187→
   188→      // Validate file size (5MB)
   189→      if (file.size > 5 * 1024 * 1024) {
   190→        onUpdate(traveler.id, 'ocrError', 'File size exceeds 5 MB limit');
   191→        onUpdate(traveler.id, 'ocrStatus', 'error');
   192→        return;
   193→      }
   194→
   195→      // Update file name
   196→      onUpdate(traveler.id, 'passportFileName', file.name);
   197→      onUpdate(traveler.id, 'ocrStatus', 'scanning');
   198→      onUpdate(traveler.id, 'ocrError', '');
   199→
   200→      try {
   201→        const base64 = await fileToBase64(file);
   202→        const res = await fetch('/api/ocr', {
   203→          method: 'POST',
   204→          headers: { 'Content-Type': 'application/json' },
   205→          body: JSON.stringify({ imageBase64: base64, documentType: 'passport' }),
   206→        });
   207→        const data = await res.json();
   208→
   209→        if (data.success && data.fields) {
   210→          for (const f of data.fields) {
   211→            if (f.value) {
   212→              const key = f.field as keyof TravelerData;
   213→              if (key === 'dateOfBirth' || key === 'dateOfIssue' || key === 'dateOfExpiry') {
   214→                const formatted = formatDateForInput(f.value);
   215→                if (formatted) onUpdate(traveler.id, key, formatted);
   216→              } else {
   217→                onUpdate(traveler.id, key, f.value);
   218→              }
   219→            }
   220→          }
   221→          onUpdate(traveler.id, 'ocrStatus', 'done');
   222→        } else {
   223→          onUpdate(traveler.id, 'ocrError', data.error || 'OCR failed. Please enter details manually.');
   224→          onUpdate(traveler.id, 'ocrStatus', 'error');
   225→        }
   226→      } catch (err) {
   227→        console.error('OCR upload error:', err);
   228→        onUpdate(traveler.id, 'ocrError', 'Network error. Please try again.');
   229→        onUpdate(traveler.id, 'ocrStatus', 'error');
   230→      }
   231→
   232→      // Reset file input
   233→      if (passportInputRef.current) passportInputRef.current.value = '';
   234→    },
   235→    [traveler.id, onUpdate]

   616→/* ─── Helper: Convert File to Base64 ─── */
   617→function fileToBase64(file: File): Promise<string> {
   618→  return new Promise((resolve, reject) => {
   619→    const reader = new FileReader();
   620→    reader.onload = () => {
   621→      const result = reader.result as string;
   622→      // Remove the data URL prefix
   623→      const base64 = result.split(',')[1];
   624→      resolve(base64);
   625→    };
   626→    reader.onerror = reject;
   627→    reader.readAsDataURL(file);
   628→  });
   629→}

   631→/* ─── Progress Stepper Component ─── */
   632→function ProgressStepper({ currentStep }: { currentStep: number }) {
   633→  const steps = ['Application Type', 'Internal ID', 'Traveler Details', 'Additional Documents', 'Review', 'Submit'];
   634→

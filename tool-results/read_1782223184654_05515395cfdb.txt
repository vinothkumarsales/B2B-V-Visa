   110тЖТ   110тЖТfunction formatDateForInput(dateStr: string): string {
   111тЖТ   111тЖТ  if (!dateStr) return '';
   112тЖТ   112тЖТ  // Handle DD/MM/YYYY format from OCR
   113тЖТ   113тЖТ  const parts = dateStr.split(/[\/\-\.]/);
   114тЖТ   114тЖТ  if (parts.length === 3) {
   115тЖТ   115тЖТ    const [d, m, y] = parts.map((p) => p.trim());
   116тЖТ   116тЖТ    // If year is 2 digits, convert to 4 digits
   117тЖТ   117тЖТ    const fullYear = y.length === 2 ? (parseInt(y) > 50 ? '19' + y : '20' + y) : y;
   118тЖТ   118тЖТ    return `${fullYear}-${m.padStart(2, '0')}-${d.padStart(2, '0')}`;
   119тЖТ   119тЖТ  }
   120тЖТ   120тЖТ  return '';
   121тЖТ   121тЖТ}

   576тЖТ   576тЖТ/* тФАтФАтФА Helper: Convert File to Base64 тФАтФАтФА */
   577тЖТ   577тЖТfunction fileToBase64(file: File): Promise<string> {
   578тЖТ   578тЖТ  return new Promise((resolve, reject) => {
   579тЖТ   579тЖТ    const reader = new FileReader();
   580тЖТ   580тЖТ    reader.onload = () => {
   581тЖТ   581тЖТ      const result = reader.result as string;
   582тЖТ   582тЖТ      // Remove the data URL prefix
   583тЖТ   583тЖТ      const base64 = result.split(',')[1];
   584тЖТ   584тЖТ      resolve(base64);
   585тЖТ   585тЖТ    };
   586тЖТ   586тЖТ    reader.onerror = reject;
   587тЖТ   587тЖТ    reader.readAsDataURL(file);
   588тЖТ   588тЖТ  });
   589тЖТ   589тЖТ}
